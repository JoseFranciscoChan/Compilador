// Programa Java para ilustrar el bucle while
class whileLoopDemo
{
    public static void main(String args[])
    {
        int x = 1;
 
        // Salir cuando x llega a ser mayor que 4
        while (x <= 4)
        {
            System.out.println("Valor de x: " + x);
 
            //incrementa el valor de x para la siguiente iteración
            x++;
        }
    }
}
